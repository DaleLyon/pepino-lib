"use strict";
class NavigateStrategy {
    canGenerate(text) {
        return text.toLowerCase().startsWith("navigate to ");
    }
    generate(text) {
        var url = text.toLowerCase().replace("navigate to ", "");
        return "this.browser.url(\"" + url + "\");";
    }
}
exports.NavigateStrategy = NavigateStrategy;
