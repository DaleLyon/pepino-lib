"use strict";
class MockStrategy {
    constructor(matchingText, codeGenerated) {
        this._matchingText = matchingText;
        this._codeGenerated = codeGenerated;
    }
    canGenerate(text) {
        return text === this._matchingText;
    }
    generate(text) {
        return this._codeGenerated;
    }
}
exports.MockStrategy = MockStrategy;
