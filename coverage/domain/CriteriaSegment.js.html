"use strict";
var _ = require('underscore');
class CriteriaSegment {
    constructor(type, text) {
        this.type = type;
        this.text = text;
        this.steps = new Array();
    }
    addSteps(steps) {
        steps.forEach((s) => {
            var alreadyHasStep = _.find(this.steps, (existingStep) => {
                return existingStep.text === s.text;
            });
            var isForThisSegment = s.segment === this.text;
            if (!alreadyHasStep && isForThisSegment)
                this.steps.push(s);
        });
        return this;
    }
}
exports.CriteriaSegment = CriteriaSegment;
